cmake_minimum_required(VERSION 3.16)

project(xv6ui VERSION 0.1 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_AUTORCC ON)

# Added Core5Compat for the GraphicalEffects you are using
find_package(Qt6 REQUIRED COMPONENTS Quick Core5Compat)

qt_standard_project_setup(REQUIRES 6.8)

# Added resources.qrc here so AUTORCC compiles it automatically
qt_add_executable(appxv6ui
    main.cpp
    resources.qrc 
)

# Cleaned up the repeated keywords! 
qt_add_qml_module(appxv6ui
    URI xv6ui
    VERSION 1.0
    QML_FILES
        Main.qml
        CpuSchedulingPage.qml
        NavigationLogic.js
    RESOURCES
        test.png
)

# Qt for iOS sets MACOSX_BUNDLE_GUI_IDENTIFIER automatically since Qt 6.1.
set_target_properties(appxv6ui PROPERTIES
    MACOSX_BUNDLE_BUNDLE_VERSION ${PROJECT_VERSION}
    MACOSX_BUNDLE_SHORT_VERSION_STRING ${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}
    MACOSX_BUNDLE TRUE
    WIN32_EXECUTABLE TRUE
)

# Added Qt6::Core5Compat to the linker
target_link_libraries(appxv6ui
    PRIVATE 
        Qt6::Quick
        Qt6::Core5Compat
)

include(GNUInstallDirs)
install(TARGETS appxv6ui
    BUNDLE DESTINATION .
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)
